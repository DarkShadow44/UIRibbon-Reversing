using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO;
using System.Diagnostics;

namespace ConsoleApplication1
{
    class Program
    {

        class Stru
        {
            public string strLarge;
            public string strSmall;

            public string infoLarge;
            public string infoSmall;

            public Stru(string strLarge, string strSmall, string infoLarge, string infoSmall)
            {
                this.strLarge = strLarge;
                this.strSmall = strSmall;
                this.infoLarge = infoLarge;
                this.infoSmall = infoSmall;
            }

        }

        static Stru GetSize(int i)
        {
            switch (i)
            {
                case 0:
                    return new Stru("Large", "Large", "large", "large");
                case 1:
                    return new Stru("Small", "Large", "small", "large");
                case 2:
                    return new Stru("Small", "Small", "small", "small");
                case 3:
                    return new Stru("Large", "Small", "large", "small");
            }
            return null;
        }

        static Stru GetLabel(int i)
        {
            switch (i)
            {
                case 0:
                    return new Stru("true", "true", "labelvisible", "labelvisible");
                case 1:
                    return new Stru("false", "true", "labelhidden", "labelvisible");
                case 2:
                    return new Stru("true", "false", "labelvisible", "labelhidden");
                case 3:
                    return new Stru("false", "false", "labelhidden", "labelhidden");
            }
            return null;
        }

        static Stru GetImage(int i)
        {
            switch (i)
            {
                case 0:
                    return new Stru("true", "true", "imagevisible", "imagevisible");
                case 1:
                    return new Stru("false", "true", "imagehidden", "imagevisible");
                case 2:
                    return new Stru("true", "false", "imagevisible", "imagehidden");
                case 3:
                    return new Stru("false", "false", "imagehidden", "imagehidden");
            }
            return null;
        }

        static void Main(string[] args)
        {
            string str = File.ReadAllText(@"E:\_Ramdisk\2.txt");

            for (int i = 0; i < 4; i++)
            {
                for (int j = 0; j < 4; j++)
                {
                    for (int k = 0; k < 4; k++)
                    {
                        var size = GetSize(i);
                        var label = GetLabel(j);
                        var image = GetImage(k);
                        string format = string.Format(str, size.strLarge, size.strSmall, label.strLarge, label.strSmall, image.strLarge, image.strSmall);
                        string filename = string.Format(@"E:\_Ramdisk\sizelarge_sizeother__{0}_{1}__{2}_{3}__{4}_{5}.xml", size.infoLarge, size.infoSmall, label.infoLarge, label.infoSmall, image.infoLarge, image.infoSmall);
                        File.WriteAllText( filename, format);

                        var filenamedest = Path.ChangeExtension(filename, "bml");

                        var info = new ProcessStartInfo();
                        info.Arguments = string.Format("{0} {1}", filename, filenamedest);
                        info.FileName = @"C:\UICC.exe";
                        Process.Start(info).WaitForExit();

                        if (!File.Exists(filenamedest))
                            File.Delete(filename);
                    }
                }
            }

        }
    }
}
